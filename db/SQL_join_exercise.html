Q1사원번호, 이름, 업무, 부서번호, 부서명, 근무지
-- 방법1: WHERE 절
SELECT  e.empno, e.ename, e.job, e.deptno,
        d.dname, d.loc
FROM    emp e, dept d
WHERE   e.deptno = d.deptno;

-- 방법2: INNER JOIN
SELECT  e.empno, e.ename, e.job, e.deptno,
        d.dname, d.loc
FROM    emp e  INNER JOIN  dept d
ON      e.deptno = d.deptno;
Q2SALESMAN 사원만 사원번호, 이름, 급여, 부서명, 근무지
SELECT  e.empno, e.ename, e.sal,
        d.dname, d.loc
FROM    emp e, dept d
WHERE   e.deptno = d.deptno
  AND   e.job = 'SALESMAN';
Q3보너스(comm) 받는 사원 이름, 부서명, 위치
SELECT  e.ename, d.dname, d.loc
FROM    emp e, dept d
WHERE   e.deptno = d.deptno
  AND   e.comm IS NOT NULL;
Q4부서별 부서명과 급여 합계
SELECT  d.dname,
        SUM(e.sal)  AS  sal_sum
FROM    emp e, dept d
WHERE   e.deptno = d.deptno
GROUP BY d.deptno, d.dname;
Q5MANAGER 사원의 이름, 업무, 부서명, 근무지
-- 방법1: WHERE 절
SELECT  e.ename, e.job, d.dname, d.loc
FROM    emp e, dept d
WHERE   e.deptno = d.deptno
  AND   e.job = 'MANAGER';

-- 방법2: INNER JOIN
SELECT  e.ename, e.job, d.dname, d.loc
FROM    emp e  INNER JOIN  dept d
ON      e.deptno = d.deptno
WHERE   e.job = 'MANAGER';

Q6관리자보다 먼저 입사한 사원 (SELF JOIN)
SELECT  e1.ename      AS  emp_name,
        e1.hiredate   AS  emp_hiredate,
        e2.ename      AS  mgr_name,
        e2.hiredate   AS  mgr_hiredate
FROM    emp e1, emp e2
WHERE   e1.mgr = e2.empno
  AND   e1.hiredate < e2.hiredate;
Q7이름에 'L'이 있는 사원 이름, 업무, 부서명, 위치
-- 방법1: INNER JOIN
SELECT  e.ename, e.job, d.dname, d.loc
FROM    emp e  INNER JOIN  dept d
ON      e.deptno = d.deptno
WHERE   e.ename LIKE '%L%';

-- 방법2: WHERE 절
SELECT  e.ename, e.job, d.dname, d.loc
FROM    emp e, dept d
WHERE   e.deptno = d.deptno
  AND   e.ename LIKE '%L%';
Q8KING 포함 모든 사원 출력 (LEFT OUTER JOIN)
SELECT  e1.empno   AS  empno,
        e1.ename   AS  ename,
        e1.mgr     AS  mgr_no,
        e2.ename   AS  mgr_name
FROM    emp e1  LEFT OUTER JOIN  emp e2
ON      e1.mgr = e2.empno;

Q9성이 'King'인 사원의 사번과 부서명
SELECT  e.employee_id,
        e.last_name,
        d.department_name
FROM    employees e  INNER JOIN  departments d
ON      e.department_id = d.department_id
WHERE   e.last_name = 'King';
Q10성이 'King'인 사원의 사번, 부서명, 업무명 (3테이블 조인)
SELECT  e.employee_id,
        e.last_name,
        d.department_name,
        j.job_title
FROM    employees e
        INNER JOIN  departments d  ON  e.department_id = d.department_id
        INNER JOIN  jobs j          ON  e.job_id = j.job_id
WHERE   e.last_name = 'King';
Q112007년 상반기 입사 사원 (부서 미배정 포함 → LEFT OUTER JOIN)
SELECT  e.employee_id,
        e.last_name,
        e.hire_date,
        d.department_name
FROM    employees e  LEFT OUTER JOIN  departments d
ON      e.department_id = d.department_id
WHERE   YEAR(hire_date)  = 2007
  AND   MONTH(hire_date) BETWEEN 1 AND 6;
Q12Executive 부서 사원 + 매니저 이름 (King 포함 → LEFT OUTER JOIN)
-- 단계1: Executive 부서 사원 + 매니저 사번 확인
SELECT  e.last_name, e.manager_id
FROM    employees e  JOIN  departments d
ON      e.department_id = d.department_id
  AND   d.department_name = 'Executive';

-- 단계2: 매니저 이름까지 출력 (King 포함 → LEFT OUTER JOIN)
SELECT  a.last_name   AS  emp_name,
        b.last_name   AS  mgr_name,
        d.department_name
FROM    employees a
        LEFT OUTER JOIN  employees b    ON  a.manager_id = b.employee_id
        JOIN  departments d            ON  a.department_id = d.department_id
                                        AND d.department_name = 'Executive';
