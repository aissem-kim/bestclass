Q1업무별·부서별 급여합계와 인원수 (10번 제외, SALESMAN·MANAGER만)
-- 방법1: WHERE 절 활용 ⭐ 권장 (그룹화 전 필터링 → 성능 유리)
SELECT  job,
        deptno,
        SUM(sal)   AS  급여합계,
        COUNT(*)   AS  인원수
FROM    emp
WHERE   deptno <> 10
  AND   job IN ('SALESMAN', 'MANAGER')
GROUP BY job, deptno;

-- 방법2: HAVING 절 활용
SELECT  job,
        deptno,
        SUM(sal)   AS  급여합계,
        COUNT(*)   AS  인원수
FROM    emp
GROUP BY job, deptno
HAVING  deptno <> 10
  AND   job IN ('SALESMAN', 'MANAGER');
💡 WHERE vs HAVING — 개별 행 조건은 WHERE, 그룹화 결과 조건은 HAVING 사용
Q2업무별 평균급여·최대급여 (평균 2000 이상, 평균 높은 순)
SELECT  job,
        AVG(sal)   AS  avg_sal,
        MAX(sal)   AS  max_sal
FROM    emp
GROUP BY job
HAVING  AVG(sal) >= 2000
ORDER BY AVG(sal) DESC;
Q3입사년도별 인원수
SELECT  YEAR(hiredate)   AS  입사년도,
        COUNT(*)         AS  인원수
FROM    emp
GROUP BY YEAR(hiredate)
ORDER BY YEAR(hiredate);

Q1연도별 최솟값·최댓값·합계·평균
SELECT  YEAR(hire_date)          AS  hire_year,
        MIN(salary)              AS  최솟급여,
        MAX(salary)              AS  최대급여,
        SUM(salary)              AS  급여합계,
        TRUNCATE(AVG(salary), 0) AS  평균급여
FROM    employees
GROUP BY hire_year
ORDER BY hire_year;
Q2부서별 평균급여 $10,000 이상 (높은 순)
SELECT  department_id,
        TRUNCATE(AVG(salary), 0)  AS  avg_sal
FROM    employees
GROUP BY department_id
HAVING  AVG(salary) >= 10000
ORDER BY AVG(salary) DESC;
Q3부서별 최대급여 (10번·20번 부서)
SELECT  department_id,
        MAX(salary)   AS  최대급여
FROM    employees
WHERE   department_id IN (10, 20)
GROUP BY department_id;
