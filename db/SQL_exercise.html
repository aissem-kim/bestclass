-- 자식 테이블 생성 (모든 제약조건 포함)
CREATE TABLE sawon
(
    sabun       INT(6)
  , sawon_name  VARCHAR(15)
  , ubmu        VARCHAR(30)   DEFAULT  '개발'
  , weolgub     INT
  , buseo       INT
  , jiyeok      VARCHAR(30)   NOT NULL
  , CONSTRAINT pk_sawon_sabun  PRIMARY KEY  (sabun)
  , CONSTRAINT ck_sawon_ubmu   CHECK        (ubmu IN ('개발', '유지보수', '관리'))

Q1student 테이블 생성
-- dept 테이블의 deptno가 PK여야 참조 가능!
CREATE TABLE student
(
    no      CHAR(4)
  , name    VARCHAR(20)    NOT NULL
  , gender  VARCHAR(10)
  , addr    VARCHAR(100)
  , jumin   CHAR(14)       NOT NULL
  , deptno  INT
  , CONSTRAINT fk_student_deptno
      FOREIGN KEY (deptno) REFERENCES dept(deptno)
);
Q2no 컬럼 PRIMARY KEY 추가
ALTER TABLE student

ALTER TABLE student
    MODIFY jumin CHAR(14) NULL;
Q4addr DEFAULT '서울' 추가
ALTER TABLE student
    MODIFY addr VARCHAR(100) DEFAULT '서울';
Q5gender CHECK 제약조건 + INSERT 확인
ALTER TABLE student
    ADD CONSTRAINT ck_student_gender
    CHECK (gender IN ('남성', '여성'));

CREATE TABLE library
(
    rent_id INT
  , book_id  CHAR(5)   NOT NULL
