Q 120번 부서 사원 조회
SELECT  empno, ename, deptno
FROM    emp
WHERE   deptno = 20;
Q 2입사일 범위 조회 (BETWEEN)
SELECT  empno, ename, hiredate
FROM    emp
WHERE   hiredate BETWEEN '1981-01-01' AND '1981-06-09';
Q 3업무 조건 조회 (IN)
SELECT  ename, job
FROM    emp
WHERE   job IN ('SALESMAN', 'CLERK');
Q 4복합 조건 (AND / OR 우선순위)
-- AND가 OR보다 우선순위 높음 → 괄호로 의도 명확히!
SELECT  *
FROM    emp
WHERE   (job = 'PRESIDENT' AND sal >= 1500)
   OR   job = 'SALESMAN';
Q 5IN + AND 조합
SELECT  *
FROM    emp
WHERE   job IN ('PRESIDENT', 'SALESMAN')
  AND   sal >= 1500;
Q 6IS NULL 조건
SELECT  ename, sal, comm
FROM    emp
WHERE   comm IS NULL;
Q 7IFNULL + ALIAS (총급여 계산)
SELECT  ename,
        sal,
        comm,
        sal + IFNULL(comm, 0)  AS  total_sal
FROM    emp;
Q 8LIKE — 첫 글자 'A'
SELECT  ename
FROM    emp
WHERE   ename LIKE 'A%';
Q 9LIKE — 두 번째 문자 'L'
SELECT  ename
FROM    emp
WHERE   ename LIKE '_L%';
Q 10LIKE — 'L'이 두 번 이상
SELECT  ename
FROM    emp
WHERE   ename LIKE '%L%L%';
Q 11IS NOT NULL 조건
SELECT  *
FROM    emp
WHERE   comm IS NOT NULL;
Q 12보너스 > 급여 × 1.1
SELECT  ename, sal, comm
FROM    emp
WHERE   comm > sal * 0.1;
Q 13IN + NOT IN 조합
SELECT  *
FROM    emp
WHERE   job IN ('CLERK', 'ANALYST')
  AND   sal NOT IN (1000, 3000, 5000);
Q 14부서 OR 관리자 조건
SELECT  *
FROM    emp
WHERE   deptno = 30
   OR   mgr = 7782;

Q 1직업코드 = 'IT_PROG'
SELECT  first_name, last_name, phone_number, job_id
FROM    EMPLOYEES
WHERE   job_id = 'IT_PROG';
Q 2부서 30 AND 급여 < 10000
SELECT  CONCAT(first_name, ' ', last_name)  AS  full_name,
        salary,
        department_id
FROM    EMPLOYEES
WHERE   department_id = 30
  AND   salary < 10000;
Q 3부서 30 AND 급여 ≤ 10000 AND 입사일 < 2006
SELECT  CONCAT(first_name, ' ', last_name)  AS  full_name
FROM    EMPLOYEES
WHERE   department_id = 30
  AND   salary <= 10000
  AND   hire_date < '2006-01-01';
Q 4두 조건 그룹 OR 조합
SELECT  *
FROM    EMPLOYEES
WHERE   (department_id = 30  AND  salary <= 10000)
   OR   (department_id = 60  AND  salary >= 5000);
Q 5BETWEEN … AND (사원번호 + 급여 범위)
SELECT  *
FROM    EMPLOYEES
WHERE   employee_id BETWEEN 110 AND 120
  AND   salary      BETWEEN 5000 AND 10000;
Q 6NOT BETWEEN (범위 제외)
SELECT  *
FROM    EMPLOYEES
WHERE   employee_id NOT BETWEEN 110 AND 120;
Q 7IN (부서번호 목록)
SELECT  *
FROM    EMPLOYEES
WHERE   department_id IN (30, 60, 90);
Q 8NOT IN (부서번호 목록 제외)
SELECT  *
FROM    EMPLOYEES
WHERE   department_id NOT IN (30, 60, 90);
Q 9LIKE — '590'으로 시작하는 전화번호
SELECT  CONCAT(first_name, ' ', last_name)  AS  full_name,
        phone_number
FROM    EMPLOYEES
WHERE   phone_number LIKE '590%';
