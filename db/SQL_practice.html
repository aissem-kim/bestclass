DDL — 테이블 생성
CREATE TABLE
-- 회원 테이블
CREATE TABLE ex_member (
    id    VARCHAR(10)   PRIMARY KEY,   -- 아이디
    pass  VARCHAR(10),                 -- 패스워드
    name  VARCHAR(20),                 -- 이름
    tel   VARCHAR(20),                 -- 전화번호
    addr  VARCHAR(100)                 -- 주소
);

-- 상품 테이블
CREATE TABLE ex_good (
    gno     VARCHAR(10)    PRIMARY KEY,   -- 상품번호
    gname   VARCHAR(30),                   -- 상품명
    gdetail VARCHAR(300),                  -- 상세설명
    price   INT                             -- 가격
);

-- 주문 테이블
CREATE TABLE ex_order (
    ono     INT           PRIMARY KEY  AUTO_INCREMENT,  -- 번호
    orderno VARCHAR(20),                                -- 주문번호
    gno     VARCHAR(10),                               -- 상품번호
    id      VARCHAR(10),                               -- 회원 아이디
    count   INT,                                       -- 개수
    status  VARCHAR(100)                               -- 배송상태
);

문제를 풀어본 후 아래 버튼으로 정답을 확인하세요.

Q1배송중인 상품 내용 (ex_order + ex_good 조인)
SELECT  o.ono, o.orderno, o.id,
        g.gname, g.price,
        o.count, o.status
FROM    ex_order o  INNER JOIN  ex_good g
ON      o.gno = g.gno
WHERE   o.status = '배송중';
Q2주문 상품 내역 + 고객 정보 (3테이블 조인)
SELECT  o.orderno,
        g.gname, g.price,
        o.count, o.status,
        m.name, m.tel, m.addr
FROM    ex_order o
        INNER JOIN  ex_good g    ON  o.gno = g.gno
        INNER JOIN  ex_member m  ON  o.id  = m.id;
Q3주문번호별 고객 + 상품 총금액
SELECT  o.orderno,
        o.id,
        SUM(g.price * o.count)  AS  total_price
FROM    ex_order o  INNER JOIN  ex_good g
ON      o.gno = g.gno
GROUP BY o.orderno, o.id
ORDER BY o.orderno;

Q4첫 번째 상품명 외 N개 형식 출력
SELECT
    o.orderno,
    CASE
        WHEN COUNT(*) = 1
            THEN MIN(g.gname)
        ELSE
            CONCAT(
                (SELECT g2.gname
                 FROM   ex_good g2
                 INNER JOIN ex_order o2 ON g2.gno = o2.gno
                 WHERE  o2.orderno = o.orderno
                 ORDER BY o2.ono
                 LIMIT 1),
                ' 외 ',
                COUNT(*) - 1,
                '개'
            )
    END  AS  order_summary,
    o.id,
    SUM(g.price * o.count)  AS  total_price
FROM    ex_order o  INNER JOIN  ex_good g
ON      o.gno = g.gno
GROUP BY o.orderno, o.id
ORDER BY o.orderno;
💡 포인트
  • LIMIT 1이 있는 서브쿼리로 첫 번째 상품명을 가져옴
  • COUNT(*) - 1로 나머지 상품 수를 계산
  • CASE WHEN COUNT(*) = 1으로 단일 주문이면 상품명만 출력
